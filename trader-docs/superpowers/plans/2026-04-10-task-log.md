**Architecture:** Spring Boot + MyBatis-Plus backend with a new `task_log` table, manual logging API via `TaskLogService`, and a Vue 3 + Element Plus frontend page to view the logs and their markdown content.

**Tech Stack:** Java 17, Spring Boot, MySQL, MyBatis-Plus, Vue 3, Element Plus.

package cc.riskswap.trader.base.dao;

import cc.riskswap.trader.admin.common.model.query.TaskLogQuery;
import cc.riskswap.trader.base.dao.entity.TaskLog;
import cc.riskswap.trader.base.dao.mapper.TaskLogMapper;
import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import org.springframework.stereotype.Repository;
import cn.hutool.core.util.StrUtil;

@Repository
public class TaskLogDao extends ServiceImpl<TaskLogMapper, TaskLog> {
    public Page<TaskLog> list(TaskLogQuery q) {
        Page<TaskLog> page = new Page<>(q.getPageNo(), q.getPageSize());
        LambdaQueryWrapper<TaskLog> wrapper = new LambdaQueryWrapper<>();
        if (StrUtil.isNotBlank(q.getTaskName())) {
            wrapper.like(TaskLog::getTaskName, q.getTaskName());
        }
        if (StrUtil.isNotBlank(q.getStatus())) {
            wrapper.eq(TaskLog::getStatus, q.getStatus());
        }
        if (q.getStartTime() != null) {
            wrapper.ge(TaskLog::getStartTime, q.getStartTime());
        }
        if (q.getEndTime() != null) {
            wrapper.le(TaskLog::getStartTime, q.getEndTime());
        }
        wrapper.orderByDesc(TaskLog::getCreatedAt);
        return this.page(page, wrapper);
    }
}
```

package cc.riskswap.trader.admin.service;

import cc.riskswap.trader.admin.common.model.dto.PageDto;
import cc.riskswap.trader.admin.common.model.dto.TaskLogDto;
import cc.riskswap.trader.admin.common.model.query.TaskLogQuery;
import cc.riskswap.trader.base.dao.TaskLogDao;
import cc.riskswap.trader.base.dao.entity.TaskLog;
import cn.hutool.core.bean.BeanUtil;
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.time.OffsetDateTime;
import java.util.List;

@Slf4j
@Service
@RequiredArgsConstructor
public class TaskLogService {
    private final TaskLogDao taskLogDao;

    public PageDto<TaskLogDto> list(TaskLogQuery query) {
        Page<TaskLog> page = taskLogDao.list(query);
        List<TaskLogDto> dtos = BeanUtil.copyToList(page.getRecords(), TaskLogDto.class);
        return new PageDto<>(dtos, page.getTotal(), query.getPageNo(), query.getPageSize());
    }

    public TaskLogDto getDetail(Long id) {
        TaskLog log = taskLogDao.getById(id);
        return BeanUtil.copyProperties(log, TaskLogDto.class);
    }

    public Long startTask(String taskName, String taskGroup) {
        TaskLog taskLog = new TaskLog();
        taskLog.setTaskName(taskName);
        taskLog.setTaskGroup(taskGroup);
        taskLog.setStartTime(OffsetDateTime.now());
        taskLog.setStatus("RUNNING");
        taskLogDao.save(taskLog);
        return taskLog.getId();
    }

    public void appendContent(Long logId, String content) {
        if (logId == null) return;
        TaskLog log = taskLogDao.getById(logId);
        if (log != null) {
            String current = log.getContent() == null ? "" : log.getContent() + "\n";
            log.setContent(current + content);
            taskLogDao.updateById(log);
        }
    }

    public void finishTask(Long logId, boolean success, String content, String errorMsg) {
        if (logId == null) return;
        TaskLog log = taskLogDao.getById(logId);
        if (log != null) {
            log.setEndTime(OffsetDateTime.now());
            log.setStatus(success ? "SUCCESS" : "FAILED");
            if (content != null) {
                String current = log.getContent() == null ? "" : log.getContent() + "\n";
                log.setContent(current + content);
            }
            log.setErrorMsg(errorMsg);
            log.setExecutionMs(log.getEndTime().toInstant().toEpochMilli() - log.getStartTime().toInstant().toEpochMilli());
            taskLogDao.updateById(log);
        }
    }
}
```
